@Test
    public void testCifrarDesplazamientoPositivo() {
        
        String textoOriginal = "Hola";
        int desplazamiento = 3;
        String resultadoEsperado = "Krod";  
        
        String resultado = CifradoCesar.cifrar(textoOriginal, desplazamiento);
        
        assertEquals(resultadoEsperado, resultado);
    }

    @Test
    public void testCifrarDesplazamientoNegativo() {
       
        String textoOriginal = "Hola";
        int desplazamiento = -1;
        String resultadoEsperado = "Gnkz";
        
        String resultado = CifradoCesar.cifrar(textoOriginal, desplazamiento);
        
        assertEquals(resultadoEsperado, resultado);
    }

    @Test
    public void testCifrarConEspacios() {
